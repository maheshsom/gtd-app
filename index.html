<script>

var SCRIPT_URL = "https://script.google.com/macros/s/AKfycbw4QPW0rIBDHZZXTM_bXg9EJFiFhxXyUOH7waGY4worMakxAFGjuPSsNN0Jx95nmKVb/exec";


document.getElementById("submitBtn").addEventListener("click", function() {

  var task = document.getElementById("task").value.trim();

  var when = document.getElementById("when").value;

  var priority = document.getElementById("priority").value;

  var dueDate = document.getElementById("dueDate").value;

  var category = document.getElementById("category").value;

  var notes = document.getElementById("notes").value.trim();

  var box = document.getElementById("status");

  var btn = document.getElementById("submitBtn");


  box.className = "status"; box.textContent = "";

  if (!task) { box.className = "status error"; box.textContent = "⚠️ Task is required."; return; }

  if (!when) { box.className = "status error"; box.textContent = "⚠️ Please select When."; return; }

  if (!priority) { box.className = "status error"; box.textContent = "⚠️ Please select Priority."; return; }

  if (!category) { box.className = "status error"; box.textContent = "⚠️ Please select Category."; return; }


  btn.disabled = true; btn.textContent = "Saving...";


  var createdOn = new Date().toLocaleDateString("en-US", { year: "numeric", month: "2-digit", day: "2-digit" });

  var row = [createdOn, task, when, priority, dueDate, category, notes];


  fetch(SCRIPT_URL, {

    method: "POST",

    mode: "no-cors",

    headers: { "Content-Type": "text/plain;charset=utf-8" },

    body: JSON.stringify({ row: row })

  })

  .then(function() {

    box.className = "status success"; box.textContent = "✅ Task added!";

    document.getElementById("task").value = "";

    document.getElementById("when").value = "";

    document.getElementById("priority").value = "";

    document.getElementById("dueDate").value = "";

    document.getElementById("category").value = "";

    document.getElementById("notes").value = "";

    btn.disabled = false; btn.textContent = "Add Task";

    setTimeout(function() { box.className = "status"; box.textContent = ""; }, 3000);

  })

  .catch(function(e) {

    box.className = "status error"; box.textContent = "⚠️ " + e.message;

    btn.disabled = false; btn.textContent = "Add Task";

  });

});

</script>
